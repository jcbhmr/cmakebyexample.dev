cmake_minimum_required(VERSION 3.0)
project(myproject)

if(WIN32)
  message("Hello Windows!")
elseif(UNIX)
  message("Hello Unix!")
endif()

find_program(clang_format clang-format)
if(clang_format)
  message("You have clang-format installed!")
endif()

execute_process(COMMAND bash --version RESULT_VARIABLE exit_code)
if(exit_code EQUAL "0")
  message("bash --version worked!")
else()
  message("bash --version failed.")
endif()
